def normalize_comuna_name(name):
    """Normalize comuna name for matching."""
    if pd.isna(name):
        return None
    name_str = str(name).upper().strip()
    replacements = {'Á': 'A', 'É': 'E', 'Í': 'I', 'Ó': 'O', 'Ú': 'U', 'Ñ': 'N'}
    for old, new in replacements.items():
        name_str = name_str.replace(old, new)
    return name_str
